\vspace{1ex}

\hspace{-3.5ex}\underline{\textbf{Определение:}} Пусть функция $f$ интегрируема на отрезке $[a; b]$, тогда выражение $ \displaystyle M[f]~=~\dfrac{1}{b-a} \int\limits_a^b f(x) \, dx$ называется \emph{средним значением функции $f$ на $[a; b]$.}
\index{Среднее значение функции}

\vspace{2ex}

\hspace{-3.5ex}\underline{\textbf{Утверждение:}} Если функция $f$ непрерывна на отрезке $[a; b],$ то $\exists \xi \in (a; b),$ такое что $M[f]~=~f(\xi).$

\vspace{2ex}

\hspace{-3.5ex}\underline{\textbf{Теорема}} \emph{(Первая теорема о среднем)}: Пусть выполнено:
\begin{enumerate}
\item $f$ и $\varphi$ интегрируемы на $[a; b]$;
\item $\varphi$ знакопостоянна на $[a; b]$;

Тогда $\displaystyle \exists \mu \in \Bigl[\inf\limits_{[a; b]}f(x); \, \sup\limits_{[a; b]}f(x)\Bigr] \,,\text{что} \; \int\limits_a^b f(x) \, \varphi(x) \, dx \, = \, \mu \int\limits_a^b  \varphi(x) \, dx.$
\item если, дополнительно, функция $f$ непрерывна на $[a; b],$ то $\exists \xi \in [a; b] \,,\text{такое что} \; \mu = f(\xi).$
\end{enumerate}
\index{Теорема!о среднем первая и вторая}
\vspace{2ex}

\hspace{-3.5ex}\underline{\textbf{Теорема}} \emph{(Вторая теорема о среднем)}: Пусть выполнено:
\begin{enumerate}
\item $f$ и $\varphi$ интегрируемы на $[a; b]$;
\item $\varphi$ монотонна на $[a; b]$;

Тогда $\displaystyle \exists \xi \in [a; b] \,,\text{что} \;  \int\limits_a^b f(x) \, \varphi(x) \, dx \, = \, \varphi(a) \int\limits_a^{\xi}  f(x) \, dx \, + \, \varphi(b) \int\limits_{\xi}^{b}  f(x) \, dx.$
\item если $\displaystyle \varphi(x) \ge 0, \;\; \varphi(x) \searrow,$ то $\displaystyle \exists \xi_1 \in [a; b] \,,\text{что} \;  \int\limits_a^b f(x) \, \varphi(x) \, dx \, = \, \varphi(a) \int\limits_a^{\xi_1}  f(x) \, dx.$
\item если $\displaystyle \varphi(x) \ge 0, \;\; \varphi(x) \nearrow,$ то $\displaystyle \exists \xi_2 \in [a; b] \,,\text{что} \;  \int\limits_a^b f(x) \, \varphi(x) \, dx \, = \, \varphi(b) \int\limits_{\xi_2}^b  f(x) \, dx.$
\end{enumerate}
\vspace{1ex}

\hspace{-3.5ex}\underline{\textbf{Замечание}}: Последние две формулы иногда называют \emph{формулами Бонне}.
\index{Формула!Бонне}
