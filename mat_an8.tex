\hspace{-3.5ex}\textbf{\underline{Утверждения:}}

\vspace{1ex}
\hspace{-3.5ex}\textbf{1.} Сходящаяся числовая последовательность не может иметь более одной предельной точки;

\vspace{0ex}
\hspace{-3.5ex}\textbf{2.} Пусть $\{x_{k_n}\}$ - подпоследовательность последовательности $\{x_n\}$. Тогда,
$$\varliminf\limits_{n \rightarrow \infty} x_n \le \varliminf\limits_{n \rightarrow \infty} x_{k_n} \;\;\; \bigl(\, \varlimsup\limits_{n \rightarrow \infty} x_n \ge \varlimsup\limits_{n \rightarrow \infty} x_{k_n}\bigr);$$

\vspace{0ex}
\hspace{-3.5ex}\textbf{3.} Если $\exists \varliminf\limits_{n \rightarrow \infty} x_n \;\; (\exists \varlimsup\limits_{n \rightarrow \infty} x_n)$, то найдется такая подпоследовательность $\{x_{k_n}\}$, что:
$$ \lim\limits_{n \rightarrow \infty} x_{k_n} = \varliminf\limits_{n \rightarrow \infty} x_n \;\;\;\;\; (\, \lim\limits_{n \rightarrow \infty} x_{k_n} = \varlimsup\limits_{n \rightarrow \infty} x_n);$$

\vspace{0ex}
\hspace{-3.5ex}\textbf{4.} $\varliminf\limits_{n \rightarrow \infty} (-x_n) = - \varlimsup\limits_{n \rightarrow \infty} x_n;$ \hspace{15ex} \textbf{5.} $\dfrac{1}{\varlimsup\limits_{n \rightarrow \infty} x_n}  = \varliminf\limits_{n \rightarrow \infty} \dfrac{1}{x_n}, \;\; x_n \neq 0$ для $\forall n.$

\begin{flushleft}\textbf{\Large{Теорема Тёплица о регулярном преобразовании последовательности}}\end{flushleft}
\index{Теорема!Тёплица о регулярном преобразовании последовательности}

Рассмотрим последовательность $A_{nm}$, зависящую от двух индексов, удовлетворяющую условиям:
\begin{enumerate}
\item $A_{nm} \geqslant 0$, для всех $n, m \in \mathds{N}$;
\item $A_{n1} + \ldots + A_{nn} = 1$;
\item $\lim\limits_{n \rightarrow \infty}A_{nm}= 0$ для любого фиксированного $m\in\mathds{N}$,

\hspace{-6ex}
\vspace{-1.5ex}
а также числовую последовательность $\{\widetilde{x}_n\}$, такую, что

\item $\exists \lim\limits_{n\rightarrow\infty}\widetilde{x}_n = a < \infty$.
\end{enumerate}

\vspace{-2ex}
Тогда последовательность $S_n=\sum\limits_{m=1}^n A_{nm} \widetilde{x}_m$ сходится и $\lim\limits_{n\rightarrow\infty} S_n=a$.

\vspace{2ex}

$\blacktriangleright$ \; Для $\forall \varepsilon>0$ рассмотрим, при достаточно больших $n$, модуль разности:
$$\left|\sum\limits_{m=1}^n A_{nm} \widetilde{x}_m - a\right| = \left\{\sum\limits_{m=1}^n A_{nm}=1\right\}=\left|\sum\limits_{m=1}^n A_{nm}(\widetilde{x}_m-a)\right|\leqslant \sum\limits_{m=1}^n A_{nm}|\widetilde{x}_m-a|  \eqno{(\ast)}$$
\emph{Получение оценок из условий 3 и 4:}

\hspace{-3.5ex}$\exists \lim\limits_{n \rightarrow \infty}\widetilde{x}_n=a\Leftrightarrow$ для $\forall\varepsilon>0$, которое было выбрано выше, $\exists N_x(\varepsilon)$:
$$\forall n > N_x(\varepsilon) \;\;\; |\widetilde{x}_n-a|<\dfrac{\varepsilon}{2} \eqno(1)$$
Далее, т.к. $\exists \lim\limits_{n \rightarrow \infty}\widetilde{x}_n=a<\infty$, то $\{\widetilde{x}_n\}$ - ограничена, т.е. $\exists M \ge |a| \ge 0$ такое, что $|\widetilde{x}_n|\leqslant M$, а
$$|\widetilde{x}_n-a|\leqslant |\widetilde{x}_n|+|a|\leqslant 2M, \; \text{для} \; \forall n\in\mathds{N} \eqno(2)$$
Так как $\lim\limits_{n \rightarrow \infty}A_{nm}=0$ и $A_{nm}\geqslant 0$, то для фиксированного нами $\varepsilon > 0  \; \exists N_A(\varepsilon) \geqslant   N_x(\varepsilon)$, что
$$0 \leqslant A_{nm}<\dfrac{\varepsilon}{4N_xM} \eqno(3)$$
для $\forall m=1, \ldots ,N_x$ и $\forall n > N_A$.

\emph{Продолжаем оценивать выражение $(\ast)$, используя неравенства (1)-(3).}

Пусть $n \geqslant N_A(\varepsilon)$ тогда
$$\sum\limits_{m=1}^{n} A_{nm}|\widetilde{x}_m-a|=A_{n1}|\widetilde{x}_1-a|+ \ldots + A_{n \scriptscriptstyle N_{x}}|\widetilde{x}_{\scriptscriptstyle N_x}-a|+A_{n \scriptscriptstyle N_{x}+1}|\widetilde{x}_{\scriptscriptstyle N_{x}+1}-a|+ \ldots + A_{nn} |\widetilde{x}-a|< $$ $$ < \; N_x \dfrac {\varepsilon}{4N_xM}\cdot 2M + \dfrac{\varepsilon}{2} \left(A_{n \scriptscriptstyle N_{x}+1} + \ldots + A_{nn} \right)< \dfrac{\varepsilon}{2} + \dfrac{\varepsilon}{2} = \varepsilon$$.
\vspace{-6ex}
\begin{flushright} $\blacksquare$ \end{flushright}
\vspace{3ex}

\hspace{-3.5ex}\textbf{\underline{Замечание:}} \; Если условие $\;\;1. \; A_{nm} \geqslant 0$, \, для $\,\forall n, m \in \mathds{N}\;\;$ не выполнено, то для справедливости теоремы Тёплица необходимо добавить условие: $\exists C > 0,$ что для $\forall n \ge 1$ следует $\displaystyle \sum\limits_{m=1}^n |A_{nm}| \, \le \, C.$
