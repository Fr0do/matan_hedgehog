\addcontentsline{toc}{section}{3. Понятие числовой последовательности}
\vspace{1ex}

\hspace{-3.5ex}\textbf{\underline{Определение:}} $\{x_n\}_{n=1}^\infty$ - множество занумерованных вещественных чисел - \textit{числовая последовательность}.
\index{Числовая последовательность}

\vspace{1ex}

\hspace{-3.5ex}\textbf{\underline{Определение:}} $\{x_n\}$ - \textit{ограничена}, если $\exists M: \forall n\in \mathds{N} \,\,\, |x_n|\leqslant M$.
\index{Числовая последовательность!ограниченная}

\hspace{13.5ex}$\{x_n\}$ - \textit{не ограничена}, если для $\forall M \;\; \exists n(M)\in \mathds{N} \; : \; |x_{n(M)}| > M$.
\index{Числовая последовательность!не ограниченная}

\vspace{1ex}

\hspace{-3.5ex}\textbf{\underline{Определение:}} $\lim\limits_{n\rightarrow\infty} x_n = a \, \Longleftrightarrow \, \forall \varepsilon > 0 \;\; \exists N(\varepsilon) \, : \, \forall n \geqslant N(\varepsilon) \;\; |x_n - a| < \varepsilon$.
\index{Предел числовой последовательности}

\vspace{1ex}

\hspace{-3.5ex}\textbf{\underline{Определение:}} $\{x_n\}$ - \textit{бесконечно малая}, если $$\lim\limits_{n\rightarrow\infty} x_n = 0 \Longleftrightarrow \forall\varepsilon > 0 \;\; \exists N(\varepsilon) \; : \; \forall n \geqslant N(\varepsilon) \;\;  |x_n|<\varepsilon.$$
\index{Числовая последовательность!бесконечно малая}

\vspace{0ex}

\hspace{-3.5ex}\textbf{\underline{Определение:}} $\{x_n\}$ - \textit{бесконечно большая}, если:
\begin{enumerate}
\item [1)] $\{x_n\}$ - знакопостоянна
\item [2)] $\lim\limits_{n\rightarrow\infty} x_n=\infty,$ то есть
$$x_n > 0 \,\,\, (x_n < 0) \,\,\, \text{для} \; \forall n \in \mathds{N} \;\; \text{и} \;\,\, \text{для} \; \forall E>0 \,\,\,\, \exists N(E): \;\; \text{при}\;\; \forall n \geqslant N(E) \,\,\,\, |x_n|>E.$$
\end{enumerate}
\index{Числовая последовательность!бесконечно большая}

\hspace{-4ex}\textbf{\underline{Замечание:}} Знакопостоянство бесконечно большой последовательности иногда не требуется.

\vspace{1ex}

\hspace{-3.5ex}\textbf{\underline{Утверждение:}} Пусть $\lim\limits_{n\rightarrow\infty} x_n = x, \lim\limits_{n\rightarrow\infty} y_n = y$, тогда
\begin{enumerate}
\item [1)] $\exists \lim\limits_{n\rightarrow\infty} (x_n\pm y_n)=x \pm y$;
\item [2)] $\exists \lim\limits_{n\rightarrow\infty} (x_n\cdot y_n)= x \cdot y$;
\item [3)] $\exists \lim\limits_{n\rightarrow\infty} \dfrac{x_n}{y_n}=\dfrac{x}{y} \,\,\,\,\,\, (y_n\neq 0 \,\,\, \text{для} \;\; \forall n, \;\; y \neq 0).$
\end{enumerate}
\index{Числовая последовательность!арифметические свойства}

\vspace{1ex}

\hspace{-3.5ex}\textbf{\underline{Утверждение:}} Если $\exists N: \,\, \forall n\geqslant N$ выполнено $x_n \leqslant y_n$, то $\lim\limits_{n\rightarrow\infty} x_n \leqslant \lim\limits_{n\rightarrow\infty} y_n$ (если данные пределы существуют).
\index{Предельный переход в неравенствах}

\task Приведите пример бесконечно малой последовательности $\{x_n\}$, члены которой:

\pnt \cw поочередно, то приближаются к своему пределу, то удаляются от него;

\pnt поочередно, то принимают значение равное своему пределу, то удаляются от него.

\vspace{3ex}

\emph{Помещённые выше примеры бесконечно малых последовательностей интересны тем, что они охватывают различные способы стремления переменной к нулю. При этом \textbf{несущественно}, лежат ли значения последовательности с одной стороны \textbf{\emph{(№3.1 б,в)}} от предела или нет \textbf{\emph{(№3.1 а,г)}}; \textbf{несущественно}, приближается ли она с каждым шагом к своему пределу \textbf{\emph{(№3.1)}} или нет \textbf{\emph{(№3.2)}}; \textbf{несущественно} наконец, достигает ли последовательность своего предела, т.е. принимает ли значения равные ему \textbf{\emph{(№3.2 б)}}. \textbf{Существенно} лишь то, о чем говорится в определении: \textbf{последовательность должна отличаться от предела сколь угодно мало для достаточно больших своих номеров}.}
